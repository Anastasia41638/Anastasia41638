using System;
using System.Data;

class Program
{
    static void Main(string[] args)
    {

        string name = "John";
        char[] letters = { 'a', 'b', 'c' };

        for (int i = 0; i < letters.Length; i++)
        {
            name += letters[i];
        }
        Console.WriteLine(name);
    }
    
}
